1. **Create `supabase/functions/_shared/sentry.ts`** (RESEARCH.md Pattern 5):
   ```typescript
   // supabase/functions/_shared/sentry.ts
   // Shared Sentry init for all Edge Functions.
   //
   // CRITICAL gotchas (RESEARCH.md Pitfall 4):
   //   1. defaultIntegrations: false  — Deno Edge Functions have no per-request scope isolation
   //   2. Callers MUST `await Sentry.flush(2000)` before returning, or events get dropped on cold-start exit
   //   3. Use withScope for per-request context, never global scope
   import * as Sentry from "npm:@sentry/deno@^10.53.1";

   Sentry.init({
     dsn: Deno.env.get("SENTRY_DSN")!,
     environment: Deno.env.get("SUPABASE_ENV") ?? "production",
     defaultIntegrations: false,
     tracesSampleRate: Deno.env.get("SUPABASE_ENV") === "production" ? 0.25 : 1.0,
   });

   export { Sentry };
   ```

2. **Create `supabase/functions/_synthetic-error/index.ts`**:
   ```typescript
   // supabase/functions/_synthetic-error/index.ts
   // Synthetic error generator. Used by Plan 07 to verify INFRA-09 / PLAT-07.
   //
   // Deploy (once SENTRY_DSN is set as a Supabase secret):
   //   supabase secrets set SENTRY_DSN="<SENTRY_DSN_EDGE from account-credentials.md>" --project-ref <STAGING_PROJECT_ID>
   //   supabase functions deploy _synthetic-error --project-ref <STAGING_PROJECT_ID>
   //
   // Invoke:
   //   curl https://<STAGING_PROJECT_ID>.supabase.co/functions/v1/_synthetic-error \
   //     -H "Authorization: Bearer <SUPABASE_ANON_KEY>"
   //
   // Expected: HTTP 500, body "logged", AND an event appears in Sentry dashboard within ~10s.
   import { Sentry } from "../_shared/sentry.ts";

   Deno.serve(async (_req) => {
     try {
       throw new Error("sentry-synthetic-edge-function-error");
     } catch (e) {
       Sentry.captureException(e);
       await Sentry.flush(2000);
       return new Response("logged", { status: 500 });
     }
   });
   ```
